    class prime
    {
        static void Main()
        {
            int p,k=0;
            Console.WriteLine("enter p");
            p = Convert.ToInt16(Console.ReadLine());
            for(int i = 1; i <= p; i++)
            {
                if (p % i == 0)
                    k++;
            }
            if (k == 2)
                Console.WriteLine("entered number is prime");
            else
                Console.WriteLine("entered num is not a prime");
        }
    }
